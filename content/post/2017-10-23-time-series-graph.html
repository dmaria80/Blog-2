library(Lahman)
library(sqldf)
library(devtools)
library(ggplot2)
# extracting the data------------------------ 
  
query<-"SELECT yearID,HR 
FROM Teams 
Where teamID='NYA'" 
  
result<-sqldf(query) 
  
#visualize the data------------------- 
  
ggplot()+ 
  geom_line(data=result,aes(x=yearID,y=HR))+ 
  xlab('year')+ 
  ylab('homeruns')
