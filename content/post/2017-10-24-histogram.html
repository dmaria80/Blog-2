library(Lahman)
library(sqldf)
library(ggplot2)
query<-"SELECT weight FROM Master"

result<-sqldf(query)

#visualizing the data

ggplot()+
  geom_histogram(data=result,aes(x=weight),color="green",fill="black",bins=50)+
  ggtitle("Baseball Player Body-Weight Distribution")
